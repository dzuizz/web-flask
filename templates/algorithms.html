Binary Search

Binary search is a search algorithm that finds the position of a target value within a sorted array.

Here is an example.

Given a sorted array of size N and a target value X, find the location of X in the array.

Input:
  N X
  A1 A2 ... AN


            
#include<iostream>
using namespace std;

int main() {
    int n, x; cin >> n >> x;
    int a[n];
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int lf = 0, rg = n-1;
    while (lf < rg) {
        int mid = (lf + rg) / 2;
        if (a[mid] >= x) {
            rg = mid;
        } else {
            lf = mid + 1;
        }
    }
    cout << lf << endl; // or rg
}

Depth First Search

Depth First Search is a graph traversal algorithm that finds the shortest path from one point to another in an unweighted graph.

Here is an example.

Given a graph of N nodes connected by M bidirectional edges which connect nodes Pi and Qi, find the shortest path from node S to node E.

Input:
  N M
  S E

  P1 Q1
  P2 Q2
  ...
  PM QM


            
#include<iostream>
#include<vector>
using namespace std;

vector<vector<int> > adj;
bool visited[n];
const int INF = 1e9;
int S, E;

int dfs(int cur) {
    visited[cur] = true;
    if (cur == E) {
        return 0;
    }
    int res = INF;
    for (int i = 0; i < adj[cur].size(); i++) {
        int next = adj[cur][i];
        if (!visited[next]) {
            int dis = dfs(next);
            if (dis != -1) {
                res = min(res, dis + 1);
            }
        }
    }
    if (res == INF) return -1;
    return res;
}

int main() {
    int n, m; cin >> n >> m;
    visited = vector<int>(n);

    for (int i = 0; i < m; i++) {
        int p, q; cin >> p >> q;
        adj[p].push_back(q);
        adj[q].push_back(p);
    }

    cin >> S >> E;
    cout << dfs(S) << endl;
}

Breadth First Search

Breadth First Search, just like Depth First Search, is a graph traversal algorithm that finds the shortest path from one point to another in an unweighted graph.

Here is an example.

Given a graph of N nodes connected by M bidirectional edges which connect nodes Pi and Qi, find the shortest path from node S to node E.

Input:
  N M
  S E

  P1 Q1
  P2 Q2
  ...
  PM QM


            
#include<iostream>
#include<queue>
using namespace std;
typedef pair<int, int> ii;

#define fi node
#define se dist

int main() {
    int n, m; cin >> n >> m;
    int s, e; cin >> s >> e;

    vector<vector<int> > adj;
    vector<bool> visited;

    for (int i = 0; i < n; i++) {
        int p, q; cin >> p >> q;
        adj[p].push_back(q);
        adj[q].push_back(p);
    }

    queue<int> q;
    q.push_back(ii(s, 0));
    visited[s] = 1;
    while(!q.empty()) {
        ii cur = q.top(); q.pop();
        if (cur.node == e) {
            cout << cur.dist << endl;
            return 0;
        }
        for (int i = 0; i < adj[cur.node].size(); i++) {
            int next = adj[cur.node][i];
            if (!visited[next]) {
                q.push_back(ii(next, cur.dist + 1));
                visited[next] = 1;
            }
        }
    }
}

Segment Tree

Segment Tree is a data structure that answers queries of a range efficiently, while being able to be modified.

Here is an example.

Given the array A with N elements find the sum of all elements from Ai to Aj.
Each query consists of 3 integers.
if the first integer is 1, the query will continue with i and j. Print the sum of all elements from Ai to Aj.
if the first integer is 2, the query will continue with U and X. Replace the element at AU with X 

Input:
  N Q
  A1 A2 ... AN

  Queries of either type at an arbitrary order.
  1 i j or
  2 U X


            
#include<iostream>
#include<vector>
using namespace std;

vector<int> segTree, a;

void build(int idx, int l, int r) {
    if (l == r) {
        return segTree[idx] = a[l]; // or a[r]
    }
    int mid = (l + r) / 2;

    build(2 * idx, l, mid);
    build(2 * idx + 1, mid + 1, r);

    segTree[idx] = segTree[2 * idx] + segTree[2 * idx + 1];
}

void update(int idx, int l, int r, int u, int x) {
    if (l == r) {
        segTree[idx] = x;
        return;
    }
    int mid = (l + r) / 2;

    if (x <= mid) update(2 * idx, l, mid, u, x);
    else update(2 * idx + 1, mid + 1, r, u, x);

    segTree[idx] = segTree[2 * idx] + segTree[2 * idx + 1];
}

int query(int idx, int l, int r, int x, int y) {
    if (l <= x && y <= r) {
        return segTree[idx];
    } else if (y <= l || x >= r) {
        return 0;
    }
    int mid = (l + r) / 2;

    return query(2 * idx, l, mid, x, y) + query(2 * idx + 1, mid + 1, r, x, y);
}

int main() {
    int n, q; cin >> n >> q;

    for (int i = 0; i < n; i++) cin >> a[i];

    segTree = vector<int>(n);

    build(1, 0, n-1);
    
}

Disjoint Set Union

Disjoint Set Union

Here is an example.

There are n kingdom. After every fight between two kingdom, the winner will take the defeated kingdom's resources
In each fight, the kingdom with more resources will win.
Given the number of kingdoms, their initial resources and the fights,
Find the kingdom with the most resources at the end.
Input:
  N M
  AN AN-1 AN-2 ... A1
  P1 Q1
  P2 Q2
  ...
  PN QN
            

            
#include<iostream>
#include<vector>
using namespace std;
typedef pair<int, int> ii;

#define fi first
#define se second

vector<int> dsu; // resources, the kingdom it is under

int find(int n) {
    if (dsu[n] == n) return n;
    return dsu[n] = find(dsu[n]);
}

void merge(int x, int y) {
    x = find(x); y = find(y);
    if (x == y) return;
    if (dsu[x].fi < dsu[y].fi) {
        swap(x, y);
    }
    dsu[x].fi += dsu[y].fi;
    dsu[y].se = x;
}

int main() {
    int n, m; cin >> n >> m;
    dsu.resize(n);

    for (int i = 0; i < n; i++) {
        cin >> dsu[i].fi;
        dsu[i].se = i;
    }

    for (int i = 0; i < m; i++) {
        int a, b; cin >> a >> b;
        merge(a, b);
    }
}
