Binary Search

Binary search is a search algorithm that finds the position of a target value within a sorted array.

Here is an example.

            
    #include<iostream>
    using namespace std;

    int main() {
        int n, x; cin >> n >> x;
        int a[n];
        for(int i = 0; i < n; i++) {
            cin >> a[i];
        }

        int lf = 0, rg = n-1;
        while (lf < rg) {
            int mid = (lf + rg) / 2;
            if (a[mid] >= x) {
                rg = mid;
            } else {
                lf = mid + 1;
            }
        }
        cout << lf << endl; // or rg
    }
