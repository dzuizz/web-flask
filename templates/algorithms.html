Binary Search

Binary search is a search algorithm that finds the position of a target value within a sorted array.

Here is an example.

Given a sorted array of size N and a target value X, find the location of X in the array.
Input:
N X
A1 A2 ... AN

            
#include<iostream>
using namespace std;

int main() {
    int n, x; cin >> n >> x;
    int a[n];
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int lf = 0, rg = n-1;
    while (lf < rg) {
        int mid = (lf + rg) / 2;
        if (a[mid] >= x) {
            rg = mid;
        } else {
            lf = mid + 1;
        }
    }
    cout << lf << endl; // or rg
}

Depth First Search

Depth First Search is a graph traversal algorithm that finds the shortest path from one point to another in an unweighted graph.

Here is an example.

Given a graph of N nodes connected by M bidirectional edges which connect nodes Pi and Qi, find the shortest path from node S to node E.
Input:
N M
A B

P1 Q1
P2 Q2
...
PM QM

            
#include<iostream>
#include<vector>
using namespace std;

vector<vector<int> > adj;
bool visited[n];
const int INF = 1e9;
int S, E;

int dfs(int cur) {
    visited[cur] = true;
    if (cur == E) {
        return 0;
    }
    int res = INF;
    for (int i = 0; i < adj[cur].size(); i++) {
        int next = adj[cur][i];
        if (!visited[next]) {
            int dis = dfs(next);
            if (dis != -1) {
                res = min(res, dis + 1);
            }
        }
    }
    if (res == INF) return -1;
    return res;
}

int main() {
    int n, m; cin >> n >> m;
    visited = vector<int>(n);

    for (int i = 0; i < m; i++) {
        int p, q; cin >> p >> q;
        adj[p].push_back(q);
        adj[q].push_back(p);
    }

    cin >> S >> E;
    cout << dfs(S) << endl;
}

Disjoint Set Union

Disjoint Set Union

Here is an example.

There are n kingdom. After every fight between two kingdom, the winner will take the defeated kingdom's resources
In each fight, the kingdom with more resources will win.
Given the number of kingdoms, their initial resources and the fights,
Find the kingdom with the most resources at the end.
Input:
N M
AN AN-1 AN-2 ... A1

P1 Q1
P2 Q2
...
PN QN

            
#include<iostream>
#include<vector>
using namespace std;
typedef pair<int, int> ii;

#define fi first
#define se second

vector<int> dsu; // resources, the kingdom it is under

int find(int n) {
    if (dsu[n] == n) return n;
    return dsu[n] = find(dsu[n]);
}

void merge(int x, int y) {
    x = find(x); y = find(y);
    if (x == y) return;
    if (dsu[x].fi < dsu[y].fi) {
        swap(x, y);
    }
    dsu[x].fi += dsu[y].fi;
    dsu[y].se = x;
}

int main() {
    int n, m; cin >> n >> m;
    dsu.resize(n);

    for (int i = 0; i < n; i++) {
        cin >> dsu[i].fi;
        dsu[i].se = i;
    }

    for (int i = 0; i < m; i++) {
        int a, b; cin >> a >> b;
        merge(a, b);
    }
}
